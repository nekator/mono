// App.tsx
import { useState } from "react";
import { DBButton, DBDrawer } from "@db-ui/react-components";

const App = () => {
	const [open, setOpen] = useState<boolean>(false);
	return (
		<div>
			<DBButton
				onClick={() => {
					setOpen(true);
				}}
			>
				Open Me
			</DBButton>
			<DBDrawer
				open={open}
				onClose={() => {
					setOpen(false);
				}}
			>
				My Drawer content
			</DBDrawer>
		</div>
	);
};

export default App;
